create or replace procedure increase_salary(empid number,percentage number)
is
esal Employees.salary%type;
begin

select salary into esal from employees
where emp_id=empid;
 esal:=esal+(esal*percentage/100);
update employees set salary=esal where emp_id=empid;

exception when no_data_found
then Dbms_output.put_line('no employee found');

end;
/

begin
increase_salary(12,5);
end;
/

CREATE OR REPLACE PROCEDURE emp_details(empid NUMBER)
IS
    ename  employees.emp_name%TYPE;
    esal   employees.salary%TYPE;
    dname  departments.dept_name%TYPE;
BEGIN
    SELECT e.emp_name, e.salary, d.dept_name
    INTO ename, esal, dname
    FROM employees e
    JOIN departments d ON e.dept_id = d.dept_id
    WHERE e.emp_id = empid;

    DBMS_OUTPUT.PUT_LINE('Name            : ' || ename);
    DBMS_OUTPUT.PUT_LINE('Salary          : ' || esal);
    DBMS_OUTPUT.PUT_LINE('Department Name : ' || dname);

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Employee not found!');
END emp_details;
/

EXEC emp_details(101);
EXEC emp_details(999);



CREATE OR REPLACE PROCEDURE process_order(porderid IN Orders1.order_id%TYPE)
IS
    product  Orders1.product_name%TYPE;
BEGIN
    SELECT product_name INTO product
    FROM Orders1
    WHERE order_id = porderid;

    UPDATE Orders1
    SET status = 'Processed'
    WHERE order_id = porderid;

    INSERT INTO orders_audits(order_id, product_name, action_date)
    VALUES(porderid, product, SYSDATE);      COMMIT;

    DBMS_OUTPUT.PUT_LINE('Order ' || porderid || ' processed successfully');

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Order ' || porderid || ' not found!');
END;
/

BEGIN
    process_order(1);
END;
/

CREATE OR REPLACE FUNCTION is_palindrome (
    pstr IN VARCHAR2
)
RETURN VARCHAR2
IS
    vrev VARCHAR2(100) := '';
BEGIN
    FOR i IN REVERSE 1..LENGTH(pstr) LOOP
        vrev := vrev || SUBSTR(pstr, i, 1);
    END LOOP;

    IF vrev = pstr THEN
        RETURN 'Yes';
    ELSE
        RETURN 'No';
    END IF;

EXCEPTION
    WHEN OTHERS THEN
        RETURN 'Error';
END;
/
select is_palindrome('aba') from dual;

FOR EACH ROW
BEGIN

CREATE or replace TRIGGER trg_after_delete_dept
after delete 
on departments
for each ROW
DECLARE
BEGIN
    INSERT INTO dept_audit (dept_id, action_date)
    VALUES (:OLD.dept_id, SYSDATE);

    INSERT INTO dept_audit (dept_id, emp_id, action_date)
